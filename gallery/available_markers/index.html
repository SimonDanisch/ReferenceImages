Available markers

using AbstractPlotting
 using GeometryTypes

 scene = Scene()
 cam2d!(scene)
 marker = collect(AbstractPlotting._marker_map)
 positions = Point2f0.(0, 1:length(marker))
 scatter!(
     scene,
     positions,
     marker = last.(marker),
     markersize = 0.8,
     raw = true,
     marker_offset = Vec2f0(0.5, -0.4)
 )
 annotations!(
     scene,
     string.(":", first.(marker)),
     positions,
     align = (:right, :center),
     textsize = 0.4,
     raw = true
 )
